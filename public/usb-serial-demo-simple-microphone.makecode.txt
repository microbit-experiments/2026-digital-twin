function emitSensors() {
    serial.writeLine('{"type":"accelerometer","x":' + input.acceleration(Dimension.X) + ',"y":' + input.acceleration(Dimension.Y) + ',"z":' + input.acceleration(Dimension.Z) + '}')
    serial.writeLine('{"type":"magnetometer","x":' + input.magneticForce(Dimension.X) + ',"y":' + input.magneticForce(Dimension.Y) + ',"z":' + input.magneticForce(Dimension.Z) + '}')
    serial.writeLine('{"type":"temperature","celsius":' + input.temperature() + '}')
}

function sendButton(button: string, event: string, label: string, value?: number) {
    if (value == null) {
        serial.writeLine('{"type":"button","button":"' + button + '","event":"' + event + '","label":"' + label + '"}')
    } else {
        serial.writeLine('{"type":"button","button":"' + button + '","event":"' + event + '","label":"' + label + '","value":' + value + '}')
    }
}

function sendGesture(event: string) {
    serial.writeLine('{"type":"gesture","event":"' + event + '"}')
}

function sendMicrophone(event: string) {
    serial.writeLine('{"type":"microphone","event":"' + event + '"}')
}

function clamp(v: number, minimum: number, maximum: number): number {
    if (v < minimum) return minimum
    if (v > maximum) return maximum
    return v
}

function sign(v: number): number {
    if (v > 0) return 1
    if (v < 0) return -1
    return 0
}

function plotPoint(x: number, y: number) {
    if (x < 0 || x > 4 || y < 0 || y > 4) return
    led.plot(x, y)
}

function drawDot() {
    basic.clearScreen()
    plotPoint(x, y)
}

function drawCross() {
    basic.clearScreen()
    plotPoint(x, y)
    plotPoint(x - 1, y)
    plotPoint(x + 1, y)
    plotPoint(x, y - 1)
    plotPoint(x, y + 1)
}

function drawBarHorizontal() {
    basic.clearScreen()
    for (let col = 0; col <= x; col += 1) {
        plotPoint(col, 2)
    }
}

function drawBarVertical() {
    basic.clearScreen()
    for (let row = 4; row >= y; row -= 1) {
        plotPoint(2, row)
    }
}

function drawSoundLevel(level: number) {
    basic.clearScreen()
    const height = clamp(Math.idiv(level, 52), 0, 4)
    for (let row = 4; row >= 4 - height; row -= 1) {
        for (let col = 1; col <= 3; col += 1) {
            plotPoint(col, row)
        }
    }
}

function drawCurrent() {
    if (demoMode === "buttons") drawCross()
    else if (demoMode === "tilt-up-down") drawDot()
    else if (demoMode === "tilt-left-right") drawDot()
    else if (demoMode === "microphone") drawSoundLevel(input.soundLevel())
    else drawCross()
}

function showRandomIcon() {
    const pick = Math.randomRange(0, 5)
    if (pick === 0) basic.showIcon(IconNames.Heart)
    else if (pick === 1) basic.showIcon(IconNames.Happy)
    else if (pick === 2) basic.showIcon(IconNames.Silly)
    else if (pick === 3) basic.showIcon(IconNames.Diamond)
    else if (pick === 4) basic.showIcon(IconNames.Target)
    else basic.showIcon(IconNames.SmallSquare)
}

function sparkle() {
    basic.clearScreen()
    for (let i = 0; i < 6; i += 1) {
        plotPoint(Math.randomRange(0, 4), Math.randomRange(0, 4))
    }
    basic.pause(120)
}

function moveTo(nextX: number, nextY: number) {
    x = clamp(nextX, 0, 4)
    y = clamp(nextY, 0, 4)
    drawCurrent()
}

function buttonLeftRight(button: string) {
    if (button === "A") moveTo(x - 1, y)
    else if (button === "B") moveTo(x + 1, y)
    else {
        x = 2
        y = 2
        drawCross()
    }
}

function handleLogoShort() {
    if (demoMode === "logo") {
        showRandomIcon()
    }
}

function handleLogoLong() {
    if (demoMode === "logo") {
        const n = Math.randomRange(0, 9)
        basic.showNumber(n)
        sendButton("Logo", "value", "Random", n)
    }
}

function handleOtherEvent() {
    if (demoMode === "shake-impact") {
        sparkle()
        drawCurrent()
    }
}

function updateMicrophone() {
    if (demoMode !== "microphone") return

    const level = input.soundLevel()
    const isLoud = level >= 80
    drawSoundLevel(level)

    if (isLoud && !micWasLoud) {
        sendMicrophone("loud")
        drawSoundLevel(level)
    } else if (!isLoud && micWasLoud) {
        sendMicrophone("quiet")
    }
    micWasLoud = isLoud
}

const demoMode: string = "microphone"
let x = 2
let y = 2
let wasPressedA = false
let wasPressedB = false
let wasPressedAB = false
let logoDownAt = 0
let micWasLoud = false
let in2gWindow = false
let inTiltUpWindow = false
let inTiltDownWindow = false

input.onButtonPressed(Button.A, function () {
    sendButton("A", "click", "Click")
    if (demoMode === "buttons") {
        buttonLeftRight("A")
    }
})

input.onButtonPressed(Button.B, function () {
    sendButton("B", "click", "Click")
    if (demoMode === "buttons") {
        buttonLeftRight("B")
    }
})

input.onButtonPressed(Button.AB, function () {
    sendButton("AB", "click", "Click")
    if (demoMode === "buttons") {
        buttonLeftRight("AB")
    }
})

input.onLogoEvent(TouchButtonEvent.Touched, function () {
    logoDownAt = input.runningTime()
    sendButton("Logo", "down", "Pressed")
})

input.onLogoEvent(TouchButtonEvent.Pressed, function () {
    sendButton("Logo", "click", "Click")
})

input.onLogoEvent(TouchButtonEvent.Released, function () {
    const heldFor = input.runningTime() - logoDownAt
    sendButton("Logo", "up", "Released")
    if (heldFor > 700) {
        sendButton("Logo", "longPress", "Long press", heldFor)
        handleLogoLong()
    } else {
        sendButton("Logo", "shortPress", "Short press", heldFor)
        handleLogoShort()
    }
})

input.onSound(DetectedSound.Loud, function () {
    sendMicrophone("loud")
    if (demoMode === "microphone") {
        drawSoundLevel(input.soundLevel())
    }
})

input.onSound(DetectedSound.Quiet, function () {
    sendMicrophone("quiet")
})

input.onGesture(Gesture.Shake, function () {
    sendGesture("shake")
    if (demoMode === "shake-impact") {
        handleOtherEvent()
    }
})

input.onGesture(Gesture.TiltLeft, function () {
    sendGesture("tiltLeft")
    if (demoMode === "tilt-left-right") {
        moveTo(x - 1, y)
    }
})

input.onGesture(Gesture.TiltRight, function () {
    sendGesture("tiltRight")
    if (demoMode === "tilt-left-right") {
        moveTo(x + 1, y)
    }
})

input.onGesture(Gesture.LogoUp, function () {
    sendGesture("faceUp")
})

input.onGesture(Gesture.LogoDown, function () {
    sendGesture("faceDown")
})

input.onGesture(Gesture.FreeFall, function () {
    sendGesture("freefall")
    if (demoMode === "shake-impact") {
        handleOtherEvent()
    }
})

input.onGesture(Gesture.ThreeG, function () {
    sendGesture("acceleration3g")
    if (demoMode === "shake-impact") {
        handleOtherEvent()
    }
})

input.onGesture(Gesture.SixG, function () {
    sendGesture("acceleration6g")
    if (demoMode === "shake-impact") {
        handleOtherEvent()
    }
})

input.onGesture(Gesture.EightG, function () {
    sendGesture("acceleration8g")
    if (demoMode === "shake-impact") {
        handleOtherEvent()
    }
})

serial.onDataReceived(serial.delimiters(Delimiters.NewLine), function () {
    const command = serial.readLine()
    if (command.indexOf('"type":"show_text"') >= 0) {
        basic.showString('ok')
    } else if (command.indexOf('"type":"clear_display"') >= 0) {
        basic.clearScreen()
    } else if (command.indexOf('"type":"set_led"') >= 0) {
        led.plot(2, 2)
    }
})

input.setSoundThreshold(SoundThreshold.Loud, 80)
input.setSoundThreshold(SoundThreshold.Quiet, 45)

if (demoMode === "logo") showRandomIcon()
else drawCurrent()

basic.forever(function () {
    const isPressedA = input.buttonIsPressed(Button.A)
    if (isPressedA && !wasPressedA) sendButton("A", "down", "Pressed")
    else if (!isPressedA && wasPressedA) sendButton("A", "up", "Released")
    wasPressedA = isPressedA

    const isPressedB = input.buttonIsPressed(Button.B)
    if (isPressedB && !wasPressedB) sendButton("B", "down", "Pressed")
    else if (!isPressedB && wasPressedB) sendButton("B", "up", "Released")
    wasPressedB = isPressedB

    const isPressedAB = input.buttonIsPressed(Button.AB)
    if (isPressedAB && !wasPressedAB) sendButton("AB", "down", "Pressed")
    else if (!isPressedAB && wasPressedAB) sendButton("AB", "up", "Released")
    wasPressedAB = isPressedAB

    const pitch = input.rotation(Rotation.Pitch)
    if (pitch > 70 && !inTiltUpWindow) {
        sendGesture("tiltUp")
        if (demoMode === "tilt-up-down") moveTo(x, y + 1)
    }
    if (pitch < -70 && !inTiltDownWindow) {
        sendGesture("tiltDown")
        if (demoMode === "tilt-up-down") moveTo(x, y - 1)
    }
    inTiltUpWindow = pitch > 50
    inTiltDownWindow = pitch < -50

    const magnitude = Math.max(
        Math.max(Math.abs(input.acceleration(Dimension.X)), Math.abs(input.acceleration(Dimension.Y))),
        Math.abs(input.acceleration(Dimension.Z))
    )
    const nowStrong = magnitude >= 1800
    if (nowStrong && !in2gWindow) {
        sendGesture("acceleration2g")
        if (demoMode === "shake-impact") handleOtherEvent()
    }
    in2gWindow = nowStrong

    updateMicrophone()
    emitSensors()
    basic.pause(200)
})
