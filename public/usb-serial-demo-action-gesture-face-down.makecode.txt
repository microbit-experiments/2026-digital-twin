function emitSensors() {
    serial.writeLine('{"type":"accelerometer","x":' + input.acceleration(Dimension.X) + ',"y":' + input.acceleration(Dimension.Y) + ',"z":' + input.acceleration(Dimension.Z) + '}')
    serial.writeLine('{"type":"magnetometer","x":' + input.magneticForce(Dimension.X) + ',"y":' + input.magneticForce(Dimension.Y) + ',"z":' + input.magneticForce(Dimension.Z) + '}')
    serial.writeLine('{"type":"temperature","celsius":' + input.temperature() + '}')
}

function sendButton(button: string, event: string, label: string, value?: number) {
    if (value == null) {
        serial.writeLine('{"type":"button","button":"' + button + '","event":"' + event + '","label":"' + label + '"}')
    } else {
        serial.writeLine('{"type":"button","button":"' + button + '","event":"' + event + '","label":"' + label + '","value":' + value + '}')
    }
}

function sendGesture(event: string) {
    serial.writeLine('{"type":"gesture","event":"' + event + '"}')
}

function sendMicrophone(event: string) {
    serial.writeLine('{"type":"microphone","event":"' + event + '"}')
}

function runForMode(mode: string, callback: () => void) {
    if (actionMode === mode) {
        callback();
    }
}

serial.onDataReceived(serial.delimiters(Delimiters.NewLine), function () {
    const command = serial.readLine()
    if (command.indexOf('"type":"show_text"') >= 0) {
        basic.showString('ok')
    } else if (command.indexOf('"type":"clear_display"') >= 0) {
        basic.clearScreen()
    } else if (command.indexOf('"type":"set_led"') >= 0) {
        led.plot(2, 2)
    }
})

basic.showIcon(IconNames.Heart)

basic.forever(function () {
    const isPressedA = input.buttonIsPressed(Button.A)
    if (isPressedA && !wasPressedA) {
        sendButton("A", "down", "Pressed")
    } else if (!isPressedA && wasPressedA) {
        sendButton("A", "up", "Released")
    }
    wasPressedA = isPressedA

    const isPressedB = input.buttonIsPressed(Button.B)
    if (isPressedB && !wasPressedB) {
        sendButton("B", "down", "Pressed")
    } else if (!isPressedB && wasPressedB) {
        sendButton("B", "up", "Released")
    }
    wasPressedB = isPressedB

    const isPressedAB = input.buttonIsPressed(Button.AB)
    if (isPressedAB && !wasPressedAB) {
        sendButton("AB", "down", "Pressed")
    } else if (!isPressedAB && wasPressedAB) {
        sendButton("AB", "up", "Released")
    }
    wasPressedAB = isPressedAB

    const x = input.acceleration(Dimension.X)
    const y = input.acceleration(Dimension.Y)
    const z = input.acceleration(Dimension.Z)
    const magnitude = Math.max(Math.max(Math.abs(x), Math.abs(y)), Math.abs(z))
    const nowStrong = magnitude >= 1800
    if (nowStrong && !in2gWindow) {
        sendGesture("acceleration2g")
        runForMode("gesture-2g", actionAccel2g)
        runForMode("default", actionAccel2g)
    }
    in2gWindow = nowStrong

    const pitch = input.rotation(Rotation.Pitch)
    if (pitch > 70 && !inTiltUpWindow) {
        sendGesture("tiltUp")
        runForMode("gesture-tilt-up-down", actionTiltUpDown)
        runForMode("default", actionTiltUpDown)
    }
    if (pitch < -70 && !inTiltDownWindow) {
        sendGesture("tiltDown")
        runForMode("gesture-tilt-up-down", actionTiltUpDown)
        runForMode("default", actionTiltUpDown)
    }
    inTiltUpWindow = pitch > 50
    inTiltDownWindow = pitch < -50

    emitSensors()
    runForMode("default", actionGesture)
    basic.pause(250)
})
